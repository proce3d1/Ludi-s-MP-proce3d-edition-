# Root = Character
API_on_character_death = {
    effect = {
        if = {
            limit = {
                has_variable = API_character_bolstered
            }
            owner = {
                set_local_variable = {
                    name  = qty_delta
                    value = -1
                }
                trigger_event = {
                    id = API_ev.2
                }
            }
        }
        if = {
            limit = {
                has_variable = API_character_suppressed
            }
            owner = {
                set_local_variable = {
                    name  = qty_delta
                    value = -1
                }
                trigger_event = {
                    id = API_ev.3
                }
            }
        }
    }
}
on_character_death = {
    on_actions = {
        API_on_character_death
    }
}
